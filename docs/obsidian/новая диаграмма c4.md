@startuml
!include <C4/C4_Container>

title Архитектура C4 Container: разрабатываемая система

LAYOUT_LEFT_RIGHT()

Person(hr, "HR / рекрутер", "Работает с откликами и результатами анализа в ATS")

System_Boundary(reqcore_system, "ATS Reqcore + модуль семантического ранжирования") {

  Container(reqcore_web, "Reqcore Web Application", "Python / Web UI", "Пользовательский интерфейс ATS для работы с вакансиями, откликами и результатами анализа")

  Container(reqcore_backend, "Reqcore Backend", "Python / REST API", "Основная серверная логика ATS. Выполняет анализ заявки и выбирает путь LLM-scoring или biencoder-scoring")

  ContainerDb(reqcore_db, "Reqcore Main Database", "PostgreSQL", "Хранит заявки, вакансии, parsedContent документов, application.score, analysis_run, criterion_score")

  Container(biencoder_api, "ATS Semantic Ranking Engine API", "Python / FastAPI", "Микросервис семантического ранжирования. Реализует /index/candidate, /search/rank, /score")

  Container(model_runtime, "Bi-encoder model runtime", "SentenceTransformer / cointegrated/rubert-tiny2", "Кодирует тексты вакансий и резюме, вычисляет cosine similarity")

  Container(candidate_index, "Candidate Index", "In-memory list", "Временный in-memory пул кандидатов CANDIDATE_INDEX для ranking API")

  Container(offline_pipeline, "Offline data & training pipeline", "Python scripts", "Подготовка данных, деидентификация, сбор пар, split train/val/test, fine-tuning модели")

  ContainerDb(training_data, "Training data artifacts", "TSV / SQLite / local files", "resumes, vacancies, HR events, features, split-файлы и артефакты обучения")
}

Rel(hr, reqcore_web, "Работает с ATS через браузер")
Rel(reqcore_web, reqcore_backend, "Вызывает REST API")
Rel(reqcore_backend, reqcore_db, "Читает и записывает данные заявок, вакансий и результатов анализа")

Rel(reqcore_backend, biencoder_api, "Вызывает /score по BIENCODER_URL для ветки provider = biencoder", "HTTP/JSON")
Rel(biencoder_api, model_runtime, "Использует модель для кодирования текстов и расчёта match_score")
Rel(biencoder_api, candidate_index, "Хранит и читает пул кандидатов для /index/candidate и /search/rank")
Rel(reqcore_backend, reqcore_db, "Сохраняет application.score и analysis_run после biencoder-scoring")

Rel(offline_pipeline, training_data, "Читает исходные данные и формирует обучающие выборки")
Rel(offline_pipeline, model_runtime, "Готовит и сохраняет fine-tuned модель")
Rel(training_data, offline_pipeline, "TSV / SQLite / local files")

UpdateRelStyle(reqcore_backend, biencoder_api, $textColor="blue", $lineColor="blue")
UpdateRelStyle(biencoder_api, model_runtime, $textColor="blue", $lineColor="blue")

SHOW_LEGEND()
@enduml
